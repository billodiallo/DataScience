from functools import reduce

#LAMBDA é uma função anônima, criada em tempo de execução.
#é aplicada uma função (LAMBDA) a uma lista de elementos, agregando o resultado até que o último elemento seja processado.
#o retorno é justamente a agregação de todas as operações executadas pela função em questão, neste caso, a LAMBDA.

array4 = [1,2,3,-20]
sub = reduce((lambda i, j: i-j), array4)
print('array original: ', array4)
print('operação com base na função REDUCE sobre o array informado -> 1-2 = -1 - 3 = -4 - (-20) = +16', '\n')
print('resultado do REDUCE sobre o array informado:', sub)

array original:  [1, 2, 3, -20]
operação com base na função REDUCE sobre o array informado -> 1-2 = -1 - 3 = -4 - (-20) = +16 

resultado do REDUCE sobre o array informado: 16

def calc_potencia8(num):
    return num ** 8

print('array original (array5):')
print(list(array5), '\n')

#list comprehension!
print('---> list comprehension <---', '\n')
print('leitura: para cada item do array5, aplicar a função calc_potencia8, e retornar o resultado em um array.')
print([calc_potencia8(item) for item in array5], '\n')

print('leitura: para cada item do array5, imprimir o próprio valor do item, se este contemplar a condição')
print('de filtragem indicada (valor>=11)')
